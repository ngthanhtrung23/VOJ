Dữ liệu
\begin{itemize}
	\item     Dòng đầu tiên chứa số nguyên dương N (1 ≤ N ≤ 10000).   
	\item     Dòng thứ hai chứa N số nguyên dương $a_{1}$    , $a_{2}$    … $a_{n}$    (1 ≤ $a_{i}$    ≤ 10000)   
	\item     Dòng thứ ba chứa N số nguyên dương $b_{1}$    , $b_{2}$    ,… $b_{n}$    (1 ≤ $b_{i}$    ≤ 10000).   
\end{itemize}
