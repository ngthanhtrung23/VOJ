   Giải thích

	\item     0 $<$ $Y_{2}$    ≤ 100   
	\item     0 $<$ 2R $<$ Y, 0 $<$ 2R $<$ $X_{2}$    – $X_{1}$
	\item     0 $<$ $P_{U}$    ≤ 100   
	\item     Luôn có một cách sắp xếp các máy bay hạ cánh.   
\end{itemize}
