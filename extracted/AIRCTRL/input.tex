Dữ liệu
\begin{itemize}
	\item     Dòng thứ nhất ghi 4 số nguyên $X_{1}$    , $X_{2}$    , $Y_{2}$    , R.   
	\item     Dòng thứ hai ghi số N.

\end{itemize}
