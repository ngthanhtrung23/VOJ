Giới hạn
\begin{itemize}
	\item     Số chuỗi lệnh nằm từ 1..50. Mỗi lệnh có từ 1 đến 50 ký tự 'S', 'L' hoặc 'R'.   
\end{itemize}
Ví dụ
\begin{verbatim}
Dữ liệu
[CASE]
$<$$<$
L
$>$$>$

[CASE]
$<$$<$
SRSL
$>$$>$

[CASE]
$<$$<$
SSSS
R
$>$$>$

[CASE]
$<$$<$
SRSL
LLSSSSSSL
SSSSSS
L
$>$$>$

[END]
Kết quả
bounded
unbounded
bounded
unbounded
\end{verbatim}
