Ví dụ
\begin{verbatim}
\textbf{Input}
3
0 1 4
3 0 2
1 4 1

\textbf{Output}
15

\textbf{Input}
4
0 1 1 1
1 0 4 3
0 1 3 5
0 0 2 5

\textbf{Output}
23

\textbf{Input}
5
4 2 2 3 3
4 2 1 4 0
1 3 4 0 1
4 3 0 2 3
0 0 3 0 4

\textbf{Output}
40\end{verbatim}

 
Giải thích
\begin{itemize}
	\item Vị trí đặt hai quân xe trong ví dụ 1 là (1, 1) và (2, 2)
	\item Vị trí đặt hai quân xe trong ví dụ 2 là (1, 3) và (1, 4)
	\item Vị trí đặt hai quân xe trong ví dụ 3 là (2, 5) và (4, 3)
\end{itemize}
Giới hạn
\begin{itemize}
	\item Có 60\% số test, 2 ≤ N ≤ 100
\end{itemize}
