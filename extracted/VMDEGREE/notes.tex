Giới hạn
\begin{itemize}
	\item 1 ≤ T ≤ 10.
	\item 1 ≤ \textbf{ N } ≤ 10 $^ 5 $ .
	\item 60\% bộ test có 1 ≤ \textbf{ N } ≤ 10 $^ 3 $ .
	\item Mỗi số trong input là một số nguyên không âm không quá 10 $^ 9 $ .
\end{itemize}
Example
\begin{verbatim}
\textbf{Input:}
1
4
2
2
1
1\end{verbatim}
\begin{verbatim}
\textbf{Output:}
YES
