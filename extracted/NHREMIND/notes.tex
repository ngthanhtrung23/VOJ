Example  
\begin{verbatim}
\textbf{Input:}
1
\\abacad
\\abbcaaa
\\
\\\textbf{Output:}
aa\end{verbatim}
   Giới hạn  
\begin{itemize}
	\item     LA = Length(A)  $\le$  1000000 (Length(A) là độ dài xâu A);   
	\item     S = Length(B[1]) + Length(B[2]) + ... + Length(B[n])  $\le$  1000000;   
	\item     Các xâu chỉ gồm chữ cái la tin 'a'..'z';   
	\item     Có 50\% số test LA, S  $\le$  100;   
\end{itemize}
