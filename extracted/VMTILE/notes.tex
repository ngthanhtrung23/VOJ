Chấm điểm

\begin{verbatim}
\textbf{Input:}
4
5 4
....
.#..
....
....
....

5 4
....
....
....
....
....

5 4
...#
....
....
....
#...

5 5
#####
.....
.....
#####
#####

\textbf{Output:}
0
95
23
8
\end{verbatim}
